  /*
    help 用来跳过两位
  */
  var help = function(arr,ind){
        var index = ind , count = 1
        while(count < 3){
            if(!arr[index]){
                count ++
                index ++
            }else{
                index ++
            }
        }
        return index
    }
    
    /*
      已有的是有序的排，要得到原有的顺序，模拟发牌的规则即可
      
      规则：（1）13张，全部背面朝上
           （2）拿上面第一张，掀开放在一边
           （3）把最上面两张，依次放在原牌堆最下面
           （4）上面的两步为一组操作
           （5）重复操作，知道牌全部掀开
           （6）此时掀开的牌，顺序为 1 ～ 13
           （7）求原来的牌的顺序
    */
    var my = function(){
        var sort = []
        for(let i = 0;i < 13;i ++){
            sort.push(false)
        }
        var count = 1;
        var ind = 0;
        /*
          count <= 13 ：即要掀开 13 张牌
          ind <= 12 : 数组的下标
        */
        while(count <= 13){
            if(ind <= 12){
                if(!sort[ind]){
                    /*
                      这里完成一组操作
                      掀开，向前进
                    */
                    sort[ind] = count
                    count ++
                    ind = help(sort,ind)
                }else{
                    /*
                      如果此时的 ind 上的牌已经掀开，则顺序往后掀开
                    */
                }else{
                    ind ++
                }
            }else{
                /*
                  下标超过 12 ，归 0
                */
                ind = 0
            }
        }
        console.log(sort)
    }
