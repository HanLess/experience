/*
    数组里是一段时期内的股价
    求：完成一次买进，卖出之后，能得到的最大的收益是多少
*/

var arr = [2,4,6,3,10,1,7,4,5,8]

var maxProfit = function(arr){
    /*
        min ：记录当前最小值
        profit : 记录当前最大收益
    */
    var min = arr[0] , profit 
    /* 
        这里可以记录从哪买进，从哪卖出
        finalMinInd : 买进点
        maxInd ：卖出点
    */
    var minInd = 0 , maxInd , finalMinInd = 0
    for(let i = 0;i < arr.length;i ++){
        if(arr[i] < min){
            min = arr[i]
            minInd = i
        }
        if(profit == undefined || arr[i] - min > profit){
            profit = arr[i] - min
            maxInd = i
            finalMinInd = minInd
        }
    }

    console.log(profit)
}

maxProfit(arr)
