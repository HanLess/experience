```
  var arr = []

  var help = function(str){
      var a = str.split("+")
      arr.push(a.map((val) => {
          return parseInt(val)
      }))
  }

  var split = function(num,max,str){
      if(num <= max){
          console.log(str + num)
          help(str + num)
          for(let i = num - 1;i > 0;i --){
              split(num - i,i,str + i + "+")
          }
      }else{
          for(let i = max;i > 0;i --){
              split(num - i,i,str + i + "+")
          }
      }
  }

  split(6,5,"")
  console.log(arr)
```
