算法导论 - 贪心算法 - 会议室问题

```
var arr = [
    [1,4],
    [3,5],
    [0,6],
    [5,7],
    [3,9],
    [5,9],
    [6,10],
    [8,11],
    [8,12],
    [2,14],
    [12,16]
]

var my = function(arr){
    var pos = arr[0]
    var result = []
    result.push(pos)
    for(let i = 1;i < arr.length;i ++){
        if(arr[i][0] >= pos[1]){
            pos = arr[i]
            result.push(pos)
        }
    }
    console.log(result)
}

my(arr)
```
